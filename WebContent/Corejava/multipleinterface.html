Multiple interface 


-->  whenever class implements interface the class must provide implementation
      to all inherited abstract method of interface otherwise the class should 
      be declare as abstract.
      
--> class can implement any number of interface.


public interface Multipleinterface
{

	void test(int arg1);
}

interface Demo3
{

	void disp(int arg2);
	
}

class Sample3 implements Multipleinterface,Demo3
{
	
	public void test(int arg1)
	{
		
		System.out.println("running test(int) method");
		System.out.println("arg1 value: "+arg1);
	}
	
	public void disp(int arg2)
	
	{
		System.out.println("running disp(int) method");
		System.out.println("arg2 value: "+arg2);
	}
	
}

class Mainclass3
{
	public static void main(String[] args)
	{
	
		System.out.println("************************");
		
		Sample3 obj1=new Sample3();
		
		obj1.test(10);
		obj1.disp(20);
		
	
		System.out.println("*************************");
	}
}
