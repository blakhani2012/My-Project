1. PRIMITIVE DATA TYPES

There are 8 primitive types

Name Size in Bits Range of Values
boolean 1 true or false
byte 8 -128 to +127
Char 16 '\u0000' to '\uFFFF'
Short 16 -32,768 to +32,767
int 32 -2,147,483,648 to +2,147,483,647
Long 64 -9,223,372,036,854,775,808 to +9,223,372,036,854,775,807
float 32 approximately 10-38 to 10+38; 7 significant digits
double 64 approximately 10-308 to 10+308; 15 significant digits

Sample Program 

package com.spy.Variable;

public class Variable 
 {  
    public static void main(String[] args)    
    {        
    
     // variable declaration      
        int a;      
        float f;     
        double d; 
     
       //variable inialization 
           
         a=12;  
         f=2.3333f;   
         d=3.33333;       
         
         // printing value of variable     
          System.out.println(a);   
          System.out.println(f);   
          System.out.println(d);     
          
         System.out.println("********************");         
         
         
         
         char c = '\u0024';     
         System.out.println(c);
         
       System.out.println("*********************");  
       
         }
         
     }
